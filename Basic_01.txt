# 1. Print Working Directory: Tells you exactly where you are right now
pwd

# 2. List: Shows all files and folders in your current location
ls

# 3. Change Directory: Moves you into a specific folder (e.g., your code project)
cd folder_name

# 4. Make Directory: Creates a new folder for your repository
mkdir my_linux_project

# 5. Create File: Creates an empty file (useful for README or scripts)
touch script.py

# 6. Remove: Deletes a file (Be careful! There is no "Recycle Bin" in terminal)
rm script.py

# 7. Super User Do: Run a command with administrative privileges
sudo apt update
